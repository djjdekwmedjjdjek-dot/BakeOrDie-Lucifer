-- โหลด Rayfield UI
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- ค่าเริ่มต้น
_G.KillAura = false
_G.AuraDistance = 25
_G.PullItems = false
_G.ESP = false
_G.WalkSpeed = 16
_G.JumpPower = 50
_G.WeaponSlot = 2

-- สร้างหน้าต่าง
local Window = Rayfield:CreateWindow({
   Name = "Bake or Die Hub",
   LoadingTitle = "Loading...",
   LoadingSubtitle = "by Lucifer",
   ConfigurationSaving = {Enabled = true, FolderName = "BakeHub"},
   KeySystem = false
})

-- ⚔️ COMBAT
local Combat = Window:CreateTab("⚔️ Combat", 4483362458)

Combat:CreateToggle({
   Name = "Kill Aura",
   CurrentValue = false,
   Callback = function(v)
      _G.KillAura = v
   end
})

Combat:CreateSlider({
   Name = "Range",
   Range = {10,50},
   Increment = 5,
   CurrentValue = 25,
   Callback = function(v)
      _G.AuraDistance = v
   end
})

-- 👤 PLAYER
local PlayerTab = Window:CreateTab("👤 Player", 4483362458)

PlayerTab:CreateSlider({
   Name = "WalkSpeed",
   Range = {16,200},
   Increment = 5,
   CurrentValue = 16,
   Callback = function(v)
      _G.WalkSpeed = v
   end
})

PlayerTab:CreateSlider({
   Name = "JumpPower",
   Range = {50,200},
   Increment = 10,
   CurrentValue = 50,
   Callback = function(v)
      _G.JumpPower = v
   end
})

-- 🎒 ITEMS
local Items = Window:CreateTab("🎒 Items", 4483362458)

Items:CreateToggle({
   Name = "Pull Items",
   CurrentValue = false,
   Callback = function(v)
      _G.PullItems = v
   end
})

-- 👁️ ESP
local ESPTab = Window:CreateTab("👁️ ESP", 4483362458)

ESPTab:CreateToggle({
   Name = "Zombie ESP",
   CurrentValue = false,
   Callback = function(v)
      _G.ESP = v
   end
})

-- 🔁 LOOP SYSTEMS

-- Kill Aura
task.spawn(function()
   while task.wait(0.2) do
      if _G.KillAura then
         local char = LocalPlayer.Character
         if char and char:FindFirstChild("HumanoidRootPart") then
            for _,z in pairs(workspace:FindFirstChild("Zombies") and workspace.Zombies:GetChildren() or {}) do
               if z:FindFirstChild("HumanoidRootPart") then
                  local dist = (z.HumanoidRootPart.Position - char.HumanoidRootPart.Position).Magnitude
                  if dist <= _G.AuraDistance then
                     pcall(function()
                        ZAP.meleeAttack.fire({
                           monsters = {z},
                           civilians = {},
                           activeSlot = _G.WeaponSlot
                        })
                     end)
                  end
               end
            end
         end
      end
   end
end)

-- WalkSpeed + Jump
task.spawn(function()
   while task.wait(1) do
      local char = LocalPlayer.Character
      if char and char:FindFirstChild("Humanoid") then
         char.Humanoid.WalkSpeed = _G.WalkSpeed
         char.Humanoid.JumpPower = _G.JumpPower
      end
   end
end)

-- Pull Items
task.spawn(function()
   while task.wait(0.5) do
      if _G.PullItems then
         local char = LocalPlayer.Character
         if char and char:FindFirstChild("HumanoidRootPart") then
            for _,v in pairs(workspace:FindFirstChild("Interactables") and workspace.Interactables:GetChildren() or {}) do
               if v:IsA("Model") and v.PrimaryPart then
                  v.PrimaryPart.CFrame = char.HumanoidRootPart.CFrame
               end
            end
         end
      end
   end
end)

-- ESP
local function addESP(obj)
   if obj:FindFirstChild("Highlight") then return end
   local h = Instance.new("Highlight")
   h.FillColor = Color3.fromRGB(255,0,0)
   h.OutlineColor = Color3.fromRGB(255,255,255)
   h.Parent = obj
end

task.spawn(function()
   while task.wait(1) do
      if _G.ESP then
         for _,z in pairs(workspace:FindFirstChild("Zombies") and workspace.Zombies:GetChildren() or {}) do
            addESP(z)
         end
      end
   end
end)

-- แจ้งเตือน
Rayfield:Notify({
   Title = "Loaded!",
   Content = "Bake or Die Hub Ready",
   Duration = 5
})
